  function updateDishes(){
    const scene = document.getElementById('orbitScene');
    const cx = scene.offsetWidth/2;
    const cy = scene.offsetHeight/2;

    dishes.forEach(d=>{
      const el = document.getElementById(d.id);
      if(!el) return;
      const ang = (aMap[d.id] + gOff) * Math.PI/180;
      const px = cx + R*Math.cos(ang) - 40;
      const py = cy + R*Math.sin(ang) - 40;

      el.style.left = px+'px';
      el.style.top  = py+'px';

      const depth = -Math.sin(ang); 
      const norm = (depth+1)/2; 
      const sc = 0.65 + norm*0.7;
      const op = 0.35 + norm*0.65;

      el.style.transform = `scale(${sc.toFixed(3)})`;
      el.style.opacity   = op.toFixed(3);
      el.style.zIndex    = Math.round(10+norm*30);
